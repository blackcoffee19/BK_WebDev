Tin học Vòng lặp For:

        int tong=0;
        for(int i=0; i<= 10; i++){
            tong = tong + i;
        }
        printf("Tổng các phần tử từ 1 => 10 %d\n",tong);

Vòng lặp While

        int tich=1;
        int i=1;
        while (i<=10)
        {
            tich = tich * i;
            i = i + 1;
        }
        printf("Tích các phần tử từ 1 => 10 %d\n", tich);
